Simulate an AR(1) + noise process: $y_t = \mu_t + \epsilon_t$ and $\mu_{t+1} = \phi \mu_t + \eta_t $ with $\phi = 0.8$

In [ ]:
phi = 0.8
mu = np.zeros(n)
y_t = np.zeros(n)

for i in range(n-1):
    mu[i+1] = phi*mu[i] + eta[i]
    y_t[i] = mu[i] + epsilon[i]
